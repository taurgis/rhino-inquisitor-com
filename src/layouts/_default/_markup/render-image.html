{{- $destination := .Destination | default "" -}} {{- $resourcePath := strings.TrimPrefix "/" $destination -}} {{- $resource := false -}} {{- if and (ne $resourcePath "") (not (or (hasPrefix $destination "http://") (hasPrefix $destination "https://"))) -}} {{- with resources.Get $resourcePath -}} {{- $resource = . -}} {{- end -}} {{- end -}} {{- $resolvedSrc := $destination -}} {{- $width := 0 -}} {{- $height := 0 -}} {{- if and $resource (reflect.IsImageResourceWithMeta $resource) -}} {{- $resolvedSrc = $resource.RelPermalink -}} {{- $width = $resource.Width -}} {{- $height = $resource.Height -}} {{- end -}}
